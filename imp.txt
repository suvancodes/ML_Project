create venv -> conda create -p venv python==3.8 -y
active venv -> conda activate venv/
initilize git -> git init
add readme file -> git add README.md
first commit -> git commit -m "first commit"
select branch -> git branch -M main
linked with github repo -> git remote add origin https://github.com/suvancodes/ML_Project.git
